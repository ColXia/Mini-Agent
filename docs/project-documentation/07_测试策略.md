本文档定义 Mini-Agent 项目的测试策略，包括测试层次、测试类型、测试工具、覆盖率要求等。

## 2. 测试层次

### 2.1 测试金字塔

```
        ┌───────────┐
        │    E2E    │  端到端测试
        │   Tests   │  (少量，高价值)
        └───────────┘
      ┌─────────────────┐
      │  Integration    │  集成测试
      │     Tests       │  (中等数量)
      └─────────────────┘
  ┌───────────────────────────┐
  │        Unit Tests         │  单元测试
  │                           │  (大量，快速)
  └───────────────────────────┘
```

| 测试类型 | 比例 | 执行时间 | 目的 |

| 单元测试 | 70% | 毫秒级 | 验证单个组件 |

## 3. 单元测试

### 3.1 测试范围

- 数据合约类
- 工具函数
- 业务逻辑服务
- 状态机转换
- 权限评估逻辑

### 3.2 测试原则

- **AAA 模式**: Arrange, Act, Assert
- **单一职责**: 每个测试只验证一个行为
- **独立性**: 测试之间无依赖
- **可重复**: 多次运行结果一致

### 3.3 测试示例

```python
import pytest
from unittest.mock import MagicMock, AsyncMock
from mini_agent.agent_core.contracts.run import Run, RunStatus
from mini_agent.application.use_cases.run_control_application_service import (
    RunControlApplicationService,
)

class TestRunControlApplicationService:
    """Tests for RunControlApplicationService."""

    @pytest.fixture
    def run_store(self):
        """Fixture for run store."""
        return MagicMock()

    @pytest.fixture
    def event_bus(self):
        """Fixture for event bus."""
        return MagicMock()

    @pytest.fixture
    def service(self, run_store, event_bus):
        """Fixture for RunControlApplicationService."""
        return RunControlApplicationService(
            run_store=run_store,
            event_bus=event_bus,
        )

    async def test_interrupt_run_success(self, service, run_store):
        """Test successful run interruption."""
        # Arrange
        run_id = "run_001"
        run = Run(
            run_id=run_id,
            agent_instance_id="agent_001",
            session_id="session_001",
            workspace_id="workspace_001",
            status=RunStatus.RUNNING,
        )
        run_store.get.return_value = run
        run_store.update = AsyncMock()

        # Act
        result = await service.interrupt_run(run_id, reason="User request")

        # Assert
        assert result.success is True
        run_store.update.assert_called_once()

    async def test_interrupt_run_not_found(self, service, run_store):
        """Test run interruption when run not found."""
        # Arrange
        run_id = "nonexistent"
        run_store.get.return_value = None

        # Act & Assert
        with pytest.raises(RunNotFoundError):
            await service.interrupt_run(run_id)
```

### 3.4 Mock 策略

```python
from unittest.mock import MagicMock, AsyncMock, patch

# 同步 Mock
mock_store = MagicMock()
mock_store.get.return_value = expected_value

# 异步 Mock
mock_async_store = MagicMock()
mock_async_store.get = AsyncMock(return_value=expected_value)

# 上下文管理器 Mock
mock_context = MagicMock()
mock_context.__enter__ = MagicMock(return_value=mock_resource)
mock_context.__exit__ = MagicMock(return_value=False)

# 补丁
with patch('module.function', return_value=mock_value):
    result = function_under_test()

import pytest
import tempfile
import os
from mini_agent.session.store import SessionStore
from mini_agent.session.entity import Session, SessionState

@pytest.mark.integration
class TestSessionStoreIntegration:
    """Integration tests for SessionStore."""

    @pytest.fixture
    def temp_dir(self):
        """Fixture for temporary directory."""
        with tempfile.TemporaryDirectory() as tmpdir:
            yield tmpdir

    @pytest.fixture
    def session_store(self, temp_dir):
        """Fixture for SessionStore."""
        return SessionStore(temp_dir)

    async def test_create_and_retrieve_session(self, session_store):
        """Test creating and retrieving a session."""
        # Arrange
        workspace_id = "workspace_001"
        agent_profile_id = "default"

        # Act
        session = await session_store.create_session(
            workspace_id=workspace_id,
            agent_profile_id=agent_profile_id,
        )
        retrieved = await session_store.get_session(session.session_id)

        # Assert
        assert retrieved is not None
        assert retrieved.session_id == session.session_id
        assert retrieved.workspace_id == workspace_id

    async def test_add_message_persists(self, session_store):
        """Test that adding message persists to storage."""
        # Arrange
        session = await session_store.create_session(
            workspace_id="workspace_001",
            agent_profile_id="default",
        )

        # Act
        await session_store.add_message(
            session.session_id,
            Message(
                message_id="msg_001",
                role=MessageRole.USER,
                content="Hello",
                timestamp=time.time(),
            ),
        )

        # Assert
        retrieved = await session_store.get_session(session.session_id)
        assert len(retrieved.messages) == 1
        assert retrieved.messages[0].content == "Hello"
```

import pytest
import subprocess
import time
import httpx

@pytest.mark.e2e
class TestCLIEndToEnd:
    """End-to-end tests for CLI."""

    def test_cli_help_command(self):
        """Test CLI help command."""
        result = subprocess.run(
            ["mini-agent", "--help"],
            capture_output=True,
            text=True,
        )
        assert result.returncode == 0
        assert "Mini-Agent" in result.stdout

    def test_cli_model_command(self):
        """Test CLI model command."""
        result = subprocess.run(
            ["mini-agent", "model", "list"],
            capture_output=True,
            text=True,
        )
        assert result.returncode == 0

@pytest.mark.e2e
class TestGatewayEndToEnd:
    """End-to-end tests for Gateway."""

    @pytest.fixture(scope="class")
    def gateway_process(self):
        """Fixture for Gateway process."""
        process = subprocess.Popen(
            ["uvicorn", "mini_agent.gateway:app", "--port", "8000"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        time.sleep(2)  # Wait for server to start
        yield process
        process.terminate()
        process.wait()

    def test_health_check(self, gateway_process):
        """Test Gateway health check."""
        response = httpx.get("http://localhost:8000/health")
        assert response.status_code == 200

    def test_list_agents(self, gateway_process):
        """Test listing agents via API."""
        response = httpx.get("http://localhost:8000/api/v1/agents")
        assert response.status_code == 200
        data = response.json()
        assert "agents" in data

## 10. 测试最佳实践

### 10.1 测试命名

```python
# 好的命名
async def test_interrupt_run_returns_success_when_run_is_running():
    ...

# 不好的命名
async def test_interrupt():
    ...
```

### 10.2 测试隔离

```python
# 好的实践：每个测试独立
@pytest.fixture
def service():
    return create_service()

async def test_case_1(service):
    ...

async def test_case_2(service):
    ...

# 不好的实践：测试之间有依赖
service = create_service()

async def test_case_1():
    service.do_something()

async def test_case_2():  # 依赖 test_case_1 的状态
    service.do_other_thing()
```

### 10.3 测试可读性

```python
# 好的实践：清晰的注释和结构
async def test_list_agents_filters_by_status():
    """Test that list_agents filters by status correctly."""
    # Arrange
    active_agent = create_agent(status="active")
    inactive_agent = create_agent(status="inactive")
    store = MockStore([active_agent, inactive_agent])
    service = AgentService(store)

    # Act
    result = await service.list_agents(status="active")

    # Assert
    assert len(result) == 1
    assert result[0].status == "active"
```
