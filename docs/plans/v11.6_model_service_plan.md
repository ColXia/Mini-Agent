### 1.2 ModelServicePort 接口定义 (拆分版)

**设计原则**: 接口拆分为三个子接口，职责更清晰

```python
# agent_core/ports/model_service_port.py

class ModelBindingPort(Protocol):
    """模型绑定相关接口"""
    
    def get_binding(self, agent_id: str) -> ModelBindingView:
        """获取 Agent 当前模型绑定"""
    
    def switch_model(self, agent_id: str, target: str) -> ModelBindingView:
        """切换模型 (支持别名/自定义名)"""
    
    def list_available_models(self) -> list[ModelOptionView]:
        """列出可用模型选项"""


class ModelClientPort(Protocol):
    """模型客户端相关接口"""
    
    def get_client(
        self, 
        binding: ModelBindingView,
        params_override: dict[str, Any] | None = None
    ) -> LLMClient:
        """获取 LLM 客户端
        
        Args:
            binding: 模型绑定视图
            params_override: 覆盖参数 (覆盖模型默认参数，未指定的继承默认值)
        
        Returns:
            已配置的 LLM 客户端
        """
    
    def get_capabilities(self, binding: ModelBindingView) -> ModelCapabilitiesView:
        """获取模型能力"""


class ModelStatusPort(Protocol):
    """模型状态查询相关接口"""
    
    def get_health_status(self, model_config_id: str) -> HealthStatusView:
        """获取模型健康状态"""
    
    def get_breaker_status(self, model_config_id: str) -> BreakerStatusView:
        """获取模型熔断状态"""
    
    def get_last_error(self, model_config_id: str) -> ErrorDetail | None:
        """获取最近一次错误详情"""
    
    def get_token_usage(self, model_config_id: str) -> TokenUsageView:
        """获取 Token 消耗情况 (用于上下文压缩判断)"""


class ModelCallbackPort(Protocol):
    """模型回调注册相关接口"""
    
    def register_breaker_callback(
        self, 
        callback: Callable[[BreakerNotification], None]
    ) -> str:
        """注册熔断触发回调
        
        Args:
            callback: 回调函数，接收 BreakerNotification
        
        Returns:
            callback_id: 用于取消注册
        """
    
    def register_health_callback(
        self,
        callback: Callable[[HealthNotification], None]
    ) -> str:
        """注册健康状态变化回调
        
        Args:
            callback: 回调函数，接收 HealthNotification
        
        Returns:
            callback_id: 用于取消注册
        """
    
    def unregister_callback(self, callback_id: str) -> bool:
        """取消注册回调
        
        Args:
            callback_id: 注册时返回的 ID
        
        Returns:
            是否成功取消
        """
    
    def clear_all_callbacks(self) -> None:
        """清除所有回调注册"""


class ModelServicePort(Protocol):
    """Agent 访问模型服务的完整接口 (组合接口)"""
    
    # 组合所有子接口
    binding: ModelBindingPort
    client: ModelClientPort
    status: ModelStatusPort
    callback: ModelCallbackPort
```

### 1.4 错误类型完整定义

```python
class ErrorType(str, Enum):
    """错误类型枚举"""
    # 网络相关
    TIMEOUT = "timeout"                    # 请求超时
    CONNECTION = "connection"              # 连接失败
    CONNECTION_REFUSED = "connection_refused"  # 连接被拒绝
    
    # 认证相关
    AUTH_FAILED = "auth_failed"            # 认证失败
    API_KEY_INVALID = "api_key_invalid"    # API Key 无效
    API_KEY_EXPIRED = "api_key_expired"    # API Key 过期
    
    # 配额相关
    QUOTA_EXCEEDED = "quota_exceeded"      # 配额耗尽
    RATE_LIMITED = "rate_limited"          # 请求频率限制
    
    # 模型相关
    MODEL_NOT_FOUND = "model_not_found"    # 模型不存在
    MODEL_OVERLOADED = "model_overloaded"  # 模型过载
    CONTEXT_LENGTH_EXCEEDED = "context_length_exceeded"  # 上下文超限
    
    # 内容相关
    CONTENT_FILTER = "content_filter"      # 内容过滤触发
    INVALID_REQUEST = "invalid_request"    # 请求格式错误
    
    # 熔断相关
    BREAKER_OPEN = "breaker_open"          # 熔断器打开
    
    # 能力相关
    CAPABILITY_UNSUPPORTED = "capability_unsupported"  # 能力不支持
    
    # 服务端错误
    SERVER_ERROR = "server_error"          # 服务端内部错误
    SERVICE_UNAVAILABLE = "service_unavailable"  # 服务不可用
    
    # 未知错误
    UNKNOWN = "unknown"


@dataclass(frozen=True)
class ErrorDetail:
    """错误详情"""
    error_type: ErrorType
    error_code: str | None      # API 返回的错误码
    error_message: str
    suggestion: str              # 用户友好的建议
    is_retryable: bool           # 是否可重试
    should_failover: bool        # 是否应故障转移
    timestamp: str
    
    # 额外信息
    retry_after: int | None      # 重试等待秒数 (rate_limit 时)
    context_used: int | None     # 已使用的上下文 (context_length_exceeded 时)
```

| 错误类型 | 可重试 | 应转移 | 处理建议 |
|---------|--------|--------|---------|
| TIMEOUT | ✓ | ✓ | 等待后重试，失败后转移 |
| CONNECTION | ✓ | ✓ | 检查网络，重试后转移 |
| CONNECTION_REFUSED | ✗ | ✓ | 检查 API Base 配置 |
| AUTH_FAILED | ✗ | ✗ | 检查 API Key 配置 |
| API_KEY_INVALID | ✗ | ✗ | 更新 API Key |
| API_KEY_EXPIRED | ✗ | ✗ | 更新 API Key |
| QUOTA_EXCEEDED | ✗ | ✓ | 切换到其他 Provider |
| RATE_LIMITED | ✓ | ✓ | 等待 retry_after 秒后重试 |
| MODEL_NOT_FOUND | ✗ | ✗ | 检查模型 ID 配置 |
| MODEL_OVERLOADED | ✓ | ✓ | 等待后重试或转移 |
| CONTEXT_LENGTH_EXCEEDED | ✗ | ✗ | 压缩上下文后重试 |
| CONTENT_FILTER | ✗ | ✗ | 修改输入内容 |
| INVALID_REQUEST | ✗ | ✗ | 检查请求格式 |
| BREAKER_OPEN | ✗ | ✓ | 等待熔断恢复或转移 |
| CAPABILITY_UNSUPPORTED | ✗ | ✗ | 使用支持该能力的模型 |
| SERVER_ERROR | ✓ | ✓ | 等待后重试 |
| SERVICE_UNAVAILABLE | ✓ | ✓ | 等待后重试或转移 |
| UNKNOWN | ✓ | ✓ | 视情况重试或转移 |

## 二、当前架构问题分析

### 2.1 Agent 侧过度暴露模型细节

**问题**: `engine.py` 直接使用 `LLMClient`，需要知道 `provider`, `api_key`, `api_base`:

```python
# 当前 engine.py 的调用方式
from mini_agent.llm.llm_wrapper import LLMClient
from mini_agent.llm.protocol_binding import ProtocolExecutionProfile

# Agent 需要知道这些细节
profile = ProtocolExecutionProfile(
    provider=LLMProvider.OPENAI,
    api_key="...",
    api_base="...",
    model="...",
)
client = LLMClient(profile=profile)
```

**目标**: Agent 只调用 `model_service.get_client(binding)`:

```python
# 目标调用方式
binding = model_service.get_binding(agent_id)
client = model_service.get_client(binding)  # 已配置好
result = await client.generate(messages, tools)
```

### 2.2 模型服务接口不完整

**问题**: `AgentModelService` 缺少以下接口:
- `get_llm_client(binding)` - Agent 无法直接获取客户端
- `get_capabilities(binding)` - Agent 需要自己解析能力字段
- 模型别名解析 - 用户无法使用 "opus", "sonnet" 等别名

### 2.3 供应商配置用户体验差

**问题**:
- 用户需要手动编辑 `providers.json`
- 无模型别名支持
- 无预设供应商快速启用
- 无能力缓存

```python
class ModelAliasResolver:
    """模型别名解析器."""

    # 预定义别名
    ALIASES: dict[str, str] = {
        "opus": "claude-opus-4-6",
        "sonnet": "claude-sonnet-4-6",
        "haiku": "claude-haiku-4-5",
        "best": "claude-opus-4-6",  # 同 opus
        "fast": "claude-haiku-4-5",  # 同 haiku
    }

    def resolve(self, alias_or_id: str) -> str:
        """解析别名到完整模型 ID."""
        normalized = alias_or_id.lower().strip()
        return self.ALIASES.get(normalized, alias_or_id)

    def is_alias(self, value: str) -> bool:
        """检查是否为别名."""
        return value.lower().strip() in self.ALIASES

    def list_aliases(self) -> list[dict[str, str]]:
        """列出所有可用别名."""
        return [
            {"alias": k, "model_id": v, "display_name": self._display_name(k)}
            for k, v in self.ALIASES.items()
        ]
```

#### Task 3.2: 供应商配置验证与诊断

```python
class ProviderConfigValidator:
    """供应商配置验证."""

    def validate(self, config: dict[str, Any]) -> ValidationResult:
        """验证供应商配置."""
        errors: list[str] = []

        # 1. 必填字段检查
        if not config.get("api_base"):
            errors.append("api_base is required")
        if not config.get("api_type"):
            errors.append("api_type is required (openai or anthropic)")

        # 2. API base 格式检查
        try:
            _normalize_api_base(config.get("api_base", ""))
        except ValueError as e:
            errors.append(str(e))

        # 3. API key 来源检查
        api_key = config.get("api_key") or os.getenv(config.get("env_key", ""))
        if not api_key and config.get("env_key"):
            errors.append(f"Environment variable {config['env_key']} not set")

        # 4. 模型列表检查
        if not config.get("models"):
            errors.append("models list is required")

        return ValidationResult(
            valid=len(errors) == 0,
            errors=errors,
            warnings=self._check_warnings(config),
        )
```

## 九、参数配置多配置设计

### 9.1 设计背景

**用户需求**: 同一个模型可以有多个参数配置，不同的 Agent 使用同一模型但拥有各自优化的参数。

**示例场景**:
```
模型: claude-sonnet-4-6
  ├── 配置A (代码助手): temperature=0.2, max_tokens=4096
  ├── 配置B (创意写作): temperature=0.8, max_tokens=8192
  └── 配置C (分析报告): temperature=0.3, response_format=json
```

### 9.2 核心设计

**参数配置独立于模型定义**:
- 每个配置有唯一 `config_id`（如 "sonnet-code"、"sonnet-creative"）
- **config_id 不跨 Provider**，每个 Provider 的模型配置独立
- 配置绑定在 `provider_id + model_id` 组合下
- Agent 绑定时绑定的是"模型+配置"的组合
- 同一模型可创建多个配置，互不影响

**配置生命周期**:
- 如果 Provider 下架某模型，关联的配置自动失效（标记为 `orphaned`）
- 失效配置不参与路由，用户可选择删除或保留

**配置数据结构**:
```python
@dataclass(frozen=True, slots=True)
class ModelParameterConfig:
    """模型参数配置 - 一个模型可以有多个配置."""
    config_id: str           # 唯一标识，如 "sonnet-code"
    model_id: str            # 模型 ID
    provider_id: str         # 提供商 ID (config_id 不跨 Provider)
    parameters: dict[str, Any]  # 参数设置
    description: str = ""    # 配置描述
    status: str = "active"   # active / orphaned (模型下架时)

    # 参数示例:
    # - temperature: float
    # - max_tokens: int
    # - stream: bool
    # - top_p: float
    # - stop_sequences: list[str]
    # - response_format: dict
    # - thinking_budget: int (扩展推理)
    # - timeout: float
```

**配置存储**: 合并到统一的 `model_registry.json`（见第十八节存储设计）

```json
{
  "configs": [
    {
      "config_id": "sonnet-code",
      "model_id": "claude-sonnet-4-6",
      "provider_id": "anthropic",
      "status": "active",
      "parameters": {
        "temperature": 0.2,
        "max_tokens": 4096,
        "stream": true
      },
      "description": "代码助手专用，低温度保证稳定"
    },
    {
      "config_id": "sonnet-creative",
      "model_id": "claude-sonnet-4-6",
      "provider_id": "anthropic",
      "status": "active",
      "parameters": {
        "temperature": 0.8,
        "max_tokens": 8192,
        "stream": true
      },
      "description": "创意写作，高温度增加多样性"
    }
  ]
}
```

### 9.3 参数优先级

```
┌────────────────────────────────────────┐
│  Agent 绑定的配置参数                   │  ← 最高优先级
```

## 十、模型默认参数配置设计

### 10.1 设计原则

**参数配置职责划分**:

| 职责 | Agent 侧 | 模型侧 |
|------|---------|--------|
| 保存 Agent 参数配置 | ✅ | ❌ |
| 保存模型默认参数 | ❌ | ✅ |
| 参数合并 | ❌ | ✅ |

**参数优先级**:

```
┌────────────────────────────────────────┐
│  Agent 传递的参数 (params_override)     │  ← 最高优先级 (覆盖)
│       ↓ 如果参数未指定                  │
│  模型默认参数                           │  ← 兜底 (继承)
└────────────────────────────────────────┘
```

**params_override 语义**:
- **覆盖模式**: Agent 传入的参数覆盖模型默认参数
- **继承模式**: Agent 未指定的参数继承模型默认值
- 示例: 模型默认 `temperature=1.0, max_tokens=4096`，Agent 传入 `temperature=0.5`，最终为 `temperature=0.5, max_tokens=4096`

### 10.2 模型默认参数配置

**存储位置**: 合并到统一的 `model_registry.json`（见第十八节存储设计）

**数据结构**:

```python

class ModelDefaultParams:

    provider_id: str
```

### 10.3 参数合并逻辑

```python
def merge_params(
    model_id: str,
    provider_id: str,
    params_override: dict[str, Any] | None
) -> dict[str, Any]:
    """合并 Agent 参数和模型默认参数"""
    
    # 1. 获取模型默认参数
    default_params = get_model_default_params(model_id, provider_id)
    
    # 2. 如果没有覆盖参数，直接返回默认参数
    if params_override is None:
        return default_params
    
    # 3. 合并参数：覆盖参数优先
    merged = {**default_params}
    for key, value in params_override.items():
        if value is not None:
            merged[key] = value
    
    return merged
```

## 十一、模型能力探测设计

### 11.1 设计原则

**核心原则**: 模型能力探测是一次性的，一个供应商下的具体型号的模型能力是固定的。

- 每个模型只需探测一次，结果持久保存
- 模型能力不会变化，除非用户手动触发重新探测
- 探测结果仅用于信息展示，不影响路由逻辑
- **探测请求不消耗 Token 额度**（使用最小化测试请求）
- **探测为单次执行**，不并发多个模型探测

### 11.2 探测触发方式

**手动触发** (用户主动发起，单次执行):


```
│                                                         │
└─────────────────────────────────────────────────────────┘
```

**错误处理代码示例**:

```python
async def probe_capability(
    model_id: str,
    provider_id: str,
    capability: str,
    probe_request: dict,
) -> ProbeResult:
    """探测单个能力"""
    try:
        response = await send_probe_request(probe_request)
        
        # 检查响应是否有效
        if is_valid_response(response):
            return ProbeResult(supported=True, status="detected")
        else:
            return ProbeResult(supported=False, status="detected")
            
    except AuthenticationError as e:
        # 认证失败，整体探测失败
        return ProbeResult(
            supported=False, 
            status="failed",
            error=ProbeError(
                type="auth_failed",
                message=str(e),
                suggestion="检查 API Key 配置"
            ),
            should_abort=True  # 终止后续探测
        )
        
    except ConnectionError as e:
        # 连接失败，整体探测失败
        return ProbeResult(
            supported=False,
            status="failed",
            error=ProbeError(
                type="connection_failed",
                message=str(e),
                suggestion="检查网络连接或 API Base 配置"
            ),
            should_abort=True
        )
        
    except TimeoutError:
        # 超时，标记为未知
        return ProbeResult(
            supported=None,  # unknown
            status="partial",
            error=ProbeError(
                type="timeout",
                message=f"探测 {capability} 超时",
                suggestion="网络可能不稳定，可稍后重试"
            )
        )
        
    except UnsupportedCapabilityError:
        # 明确不支持
        return ProbeResult(supported=False, status="detected")
        
    except Exception as e:
        # 其他错误，标记为未知
        return ProbeResult(
            supported=None,
            status="partial",
            error=ProbeError(
                type="unknown",
                message=str(e),
                suggestion="未知错误，可稍后重试"
            )
        )
```

## 十七、健康监控与熔断器

### 17.1 熔断器状态

```python
class CircuitState(str, Enum):
    CLOSED = "closed"        # 正常状态，允许请求
    OPEN = "open"           # 熔断状态，拒绝请求
    HALF_OPEN = "half_open" # 半开状态，允许探测请求
```

### 17.2 熔断器状态转换触发细节

```
┌─────────────────────────────────────────────────────────────────┐
│                    熔断器状态转换流程                            │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  ┌─────────────┐                                               │
│  │   CLOSED    │  ← 初始状态                                   │
│  │  (正常)     │    允许所有请求                               │
│  └──────┬──────┘                                               │
│         │                                                       │
│         │ 触发条件: 连续失败次数 >= failure_threshold            │
│         │ (默认 5 次)                                           │
│         │                                                       │
│         ▼                                                       │
│  ┌─────────────┐                                               │
│  │    OPEN     │  ← 熔断状态                                   │
│  │  (拒绝)     │    拒绝所有请求                               │
│  └──────┬──────┘                                               │
│         │                                                       │
│         │ 触发条件: 距离熔断时间 >= timeout_seconds              │
│         │ (默认 60 秒)                                          │
│         │                                                       │
│         ▼                                                       │
│  ┌─────────────┐                                               │
│  │  HALF_OPEN  │  ← 半开状态                                   │
│  │  (探测)     │    允许有限请求 (max 3 次)                     │
│  └──────┬──────┘                                               │
│         │                                                       │
│         ├── 触发条件: 连续成功次数 >= success_threshold          │
│         │             (默认 3 次)                               │
│         │             → 转换到 CLOSED                          │
│         │                                                       │
│         └── 触发条件: 任何失败                                   │
│                       → 转换回 OPEN                            │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

**状态转换触发条件详解**:

| 当前状态 | 触发事件 | 目标状态 | 说明 |
|---------|---------|---------|------|
| CLOSED | 连续失败 >= failure_threshold | OPEN | 熔断触发，开始计时 |
| CLOSED | 单次成功 | CLOSED | 重置失败计数 |
| CLOSED | 单次失败 | CLOSED | 失败计数 +1 |
| OPEN | 时间 >= timeout_seconds | HALF_OPEN | 允许探测请求 |
| OPEN | 时间 < timeout_seconds | OPEN | 继续拒绝 |
| HALF_OPEN | 连续成功 >= success_threshold | CLOSED | 恢复正常 |
| HALF_OPEN | 任何失败 | OPEN | 重新熔断 |
| HALF_OPEN | 探测次数 >= half_open_max_calls | HALF_OPEN | 等待结果 |

**熔断触发条件判断**:

```python
def should_trip_breaker(
    failure_count: int,
    failure_threshold: int,
    last_error: ErrorDetail,
) -> bool:
    """判断是否应该触发熔断"""
    # 连续失败次数达到阈值
    if failure_count >= failure_threshold:
        return True
    
    # 特殊错误直接触发熔断（无需等待阈值）
    immediate_trip_errors = {
        ErrorType.AUTH_FAILED,
        ErrorType.API_KEY_INVALID,
        ErrorType.API_KEY_EXPIRED,
    }
    if last_error.error_type in immediate_trip_errors:
        return True
    
    return False
```

**半开状态探测逻辑**:

```python
def handle_half_open_request(
    breaker: CircuitBreaker,
    success: bool,
) -> CircuitState:
    """处理半开状态的请求结果"""
    if success:
        breaker.success_count += 1
        if breaker.success_count >= breaker.success_threshold:
            # 恢复正常
            breaker.state = CircuitState.CLOSED
            breaker.failure_count = 0
            breaker.success_count = 0
            return CircuitState.CLOSED
    else:
        # 任何失败立即回到熔断状态
        breaker.state = CircuitState.OPEN
        breaker.failure_count += 1
        breaker.success_count = 0
        breaker.last_failure_time = time.time()
        return CircuitState.OPEN
    
    return CircuitState.HALF_OPEN
```

### 22.4 内存缓存设计

**用途**: 为上下文压缩机制提供快速 Token 查询

```python
@dataclass
class TokenUsageCache:
    """Token 使用内存缓存"""
    
    # 当前会话使用情况
    current_session: dict[str, int]  # model_config_id -> total_tokens
    
    # 最近 N 次请求的 Token 使用
    recent_usage: deque[TokenUsageRecord]
    
    # 缓存大小限制
    max_recent_records: int = 100
    
    def get_current_usage(self, model_config_id: str) -> int:
        """获取当前会话 Token 使用量"""
        return self.current_session.get(model_config_id, 0)
    
    def add_usage(self, record: TokenUsageRecord) -> None:
        """添加使用记录到缓存"""
        self.current_session[record.model_config_id] = \
            self.current_session.get(record.model_config_id, 0) + record.total_tokens
        self.recent_usage.append(record)
        
        # 超过限制时移除旧记录
        if len(self.recent_usage) > self.max_recent_records:
            self.recent_usage.popleft()
    
    def should_compress(self, model_config_id: str, threshold: float = 0.8) -> bool:
        """判断是否需要压缩上下文"""
        usage = self.get_current_usage(model_config_id)
        # 获取模型上下文窗口大小
        max_context = self._get_max_context(model_config_id)
        return usage / max_context > threshold
```

### 22.5 批量写入机制

```python
class TokenUsageWriter:
    """Token 使用批量写入器"""
    
    def __init__(self, file_path: Path, flush_interval: int = 60):
        self.file_path = file_path
        self.flush_interval = flush_interval
        self.buffer: list[TokenUsageRecord] = []
        self.last_flush = time.time()
    
    def add_record(self, record: TokenUsageRecord) -> None:
        """添加记录到缓冲区"""
        self.buffer.append(record)
        self._maybe_flush()
    
    def _maybe_flush(self) -> None:
        """检查是否需要刷新到文件"""
        if time.time() - self.last_flush >= self.flush_interval:
            self._flush()
    
    def _flush(self) -> None:
        """将缓冲区写入文件"""
        if not self.buffer:
            return
        
        # 读取现有数据
        data = self._read_existing()
        
        # 合并新记录
        data["records"].extend([r.to_dict() for r in self.buffer])
        
        # 更新汇总
        for record in self.buffer:
            key = f"{record.provider_id}:{record.model_id}"
            if key not in data["summaries"]:
                data["summaries"][key] = self._new_summary(record)
            data["summaries"][key]["total_requests"] += 1
            data["summaries"][key]["total_input_tokens"] += record.input_tokens
            data["summaries"][key]["total_output_tokens"] += record.output_tokens
            data["summaries"][key]["total_cost"] += record.total_cost
        
        # 写入文件
        self._write_file(data)
        
        # 清空缓冲区
        self.buffer.clear()
        self.last_flush = time.time()
```
